def time(x):
    value = ''
    if x == '-1':
        value = '미상'
    elif x <=5000000:
        value = '1단계'
    elif x <=10000000:
        value = '2단계'
        
    elif x <=15000000:
        value = '3단계'
    elif x <=20000000:
        value = '1단계'
    else:
        value = '5단계'
        
    return value
